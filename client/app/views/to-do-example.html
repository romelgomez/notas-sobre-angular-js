Aplicación: Lista de tareas

Directivas usadas: ng-controller ng-repeat ng-model ng-click

Se trata de una sencilla aplicación que almacena un conjunto de tareas por hacer, el usuario podrá añadir e ir marcando las tareas completadas, luego de verificar podra removerlas del listado. Vale destacar que el listado tendra un conjunto de tareas inicialmente por defecto.

Para cada tarea estan definidos los siguientes datos:
Campo descripción
texto Texto descriptivo de la tarea.
estado Estado de la tarea, sera un valor del tipo boleano es decir verdadero o falso.
Vamos por parte

Parte 1: Definimos el conjunto de tareas por defecto y mostramos en la vista cuántas son.

<script type="text/javascript">
    // Controlador
    var TareasCtrl = function($scope){

        // Definimos el conjunto de tareas por defecto.
        $scope.listaDeTareas = [
            {texto:"Aprender Angular",estado:false},
            {texto:"Estudiar Física",estado:false},
            {texto:"Estudiar Matemática",estado:false},
            {texto:"Comer",estado:true},
            {texto:"Ver el correo",estado:true}
        ];

        // Metodo
        $scope.cantidadDeTareasListadas = function(){
           return $scope.listaDeTareas.length;
        }

    }
</script>
<div ng-controller="TareasCtrl">
    <span>Lista de tareas:</span> {{cantidadDeTareasListadas()}}
</div>
Explicando la directiva:
  1. La directiva ng-controller se define el nombre del controlador que va trabajar sobre el ámbito de todo el elemento. Por convención el nombre del controlador debe ser UpperCamelCase y teminar en Ctrl.

Parte 3: Habilitamos un formulario para registrar nuevas tareas, además de crear un botón el cual nos va a permitir remover las tareas que consideramos completadas.

<script type="text/javascript">
    // Controlador
    var TareasCtrl = function($scope){

        // Definimos el conjunto de tareas por defecto.
        $scope.listaDeTareas = [
            {texto:"Aprender Angular",estado:false},
            {texto:"Estudiar Física",estado:false},
            {texto:"Estudiar Matemática",estado:false},
            {texto:"Comer",estado:true},
            {texto:"Ver el correo",estado:true}
        ];

        // Metodo
        $scope.cantidadDeTareasListadas = function(){
           return $scope.listaDeTareas.length;
        }

        // Metodo
        $scope.anadir_tarea = function(){
            // Añadimos una nueva tarea.
            $scope.listaDeTareas.push({"texto":$scope.tarea,"estado":false});
            // Reestablecemos el campo del formulario a su forma original
            $scope.tarea = '';
        }

        // Metodo
        $scope.remover_tarea = function(){
            // _.filter es una función de underscorejs, que observa cada valor de $scope.listaDeTareas, para retornar un array con todos los valores que retornan en la prueba verdadero.
            $scope.listaDeTareas = _.filter($scope.listaDeTareas,function(tarea){
                // Como !tarea.estado está negada, va a retornar verdadero cuando el estado de la tarea sea falso y viceversa.
                return !tarea.estado;
            });
        }

    }
</script>

<div ng-controller="TareasCtrl">
	<span>Lista de tareas:</span> {{cantidadDeTareasListadas()}}
	<hr>

	<ul class="unstyled">
		<li ng-repeat="tarea in listaDeTareas">
			<label class="checkbox">
				<input type="checkbox" ng-model="tarea.estado">  <span class="estado-{{tarea.estado}}" >{{tarea.texto}}</span>
			</label>
		</li>
	</ul>

	<hr>

	<form class="form-inline">
		<div class="input-prepend">
			<span class="add-on"><i class="icon-plus"></i></span><input ng-model="tarea" id="task" name="task" type="text" placeholder="Eje: Estudiar Matematica">
		</div>
		<button ng-click="anadir_tarea()" class="btn">Anotar Tarea</button>
	</form>

	<button ng-click="remover_tarea()" type="button" class="btn btn-danger"><i class="icon-warning-sign"></i> Remover las tareas completadas</button>
</div>
Explicando las directivas:
  1. La directiva ng-controller se define el nombre del controlador que va trabajar sobre el ámbito de todo el elemento. Por convención el nombre del controlador debe ser UpperCamelCase y teminar en Ctrl.
  2. La directiva ng-repeat repite todo el ámbito del elemento donde sea colocado, entre comillas esta la definición del iterador "imagen in app_images" donde app_images es el array completo que contiene todo los objetos, imagen para cada ciclo se le asignara un objeto del array, donde podras usar imagen.id, imagen.titulo, imagen.name, imagen.otro_valor aciendo saber que deben estar entre {{}} para que AngularJs los reconozca, {{imagen.id}}, {{imagen.titulo}}, {{imagen.name}}, {{imagen.otro_valor}}.
  3. La directiva ng-model es responsable de unir el modelo y la vista.
  4. La directiva ng-click permite especificar un comportamiento personalizado cuando el elemento es clicleado.
